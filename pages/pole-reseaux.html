EXEMPLE – Script Python réseau

import socket
import os

# Scanner un réseau local
def ping_host(ip):
    result = os.system(
        f"ping -c 1 -W 1 {ip}"
    )
    return result == 0

for i in range(1, 10):
    ip = f"192.168.1.{i}"
    if ping_host(ip):
        print(f"✓ {ip} actif")
